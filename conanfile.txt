[options]
# Disable FFmpeg to avoid Conan CMake error: ffmpeg target references libwebp::libwebp
# which is not found during OpenCV configure. Normitri only needs core/imgproc (resize, normalize,
# color conversion), not video I/O.
opencv/*:with_ffmpeg=False
# Disable Wayland so we don't pull xorg/system; its system_requirements() runs apt-get without sudo
# and fails on GitHub Actions (permission denied). Normitri is headless (no display).
opencv/*:with_wayland=False
